    <script>
        function checkRange()
        {
          var n1, n2, ans;
          n1 = parseInt(document.getElementById("1st number").value);
          n2 = parseInt(document.getElementById("2nd number").value);
        
          if((n1 >= 50 && n1 <= 99) || (n2 >= 50 && n2 <= 99))
          {
            ans = true;
            document.getElementById("answer").value = ans;
          }
          else
          {
            ans = false;
            document.getElementById("answer").value = ans;
          }
        }
    </script>
<p>1st Number: <input id="1st number"></p>
<p>2nd Number: <input id="2nd number"></p>
<button onclick="checkRange()">Check Range</button>
<p> The Number are in Range : <input id="answer"></p>
